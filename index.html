// Fetch with caching async function fetchWithCache(url, cacheKey) { const now = Date.now(); const cached = JSON.parse(localStorage.getItem(cacheKey) || '{}'); if (cached.time && now - cached.time < (5 + Math.random() * 3) * 3600 * 1000 && cached.data) { return cached.data; } const res = await fetch(url); const data = await res.text(); localStorage.setItem(cacheKey, JSON.stringify({ time: now, data })); return data; } // Load About Me async function loadAboutMe() { try { const markdown = await fetchWithCache('https://raw.githubusercontent.com/ivanisok/About-Me/main/README.md', 'aboutMe'); document.getElementById('aboutText').innerHTML = marked.parse(markdown); } catch (e) { console.warn('Using fallback About Me text', e); document.getElementById('aboutText').textContent = 'Hi! Im Ivan, a young dev interested in Android, I daily drive Linux! (nerd)'; } } // Load GitHub repos async function loadRepos() { try { const json = await fetchWithCache('https://api.github.com/users/ivanisok/repos?per_page=100&sort=updated', 'repos'); const repos = JSON.parse(json); const container = document.getElementById('repos'); container.innerHTML = ''; repos.forEach(repo => { const card = document.createElement('a'); card.href = repo.html_url; card.target = '_blank'; card.className = 'repoCard block bg-zinc-900 border border-zinc-800 rounded-2xl p-5 hover:border-blue-500 transition-transform duration-200 shadow hover:scale-105'; card.innerHTML = `

${repo.name}

${repo.description ?? 'No description'}

⭐ ${repo.stargazers_count} ${repo.language ?? ''}
`; container.appendChild(card); safeAnimate(card, { opacity: [0, 1], y: [20, 0] }, { duration: 0.4 }); }); } catch (e) { console.error('Failed to load GitHub repos', e); } } // Enable hover animations function enableHoverAnimations() { if (!window.matchMedia('(hover: hover)').matches) return; document.querySelectorAll('.navItem').forEach(el => { el.addEventListener('mouseenter', () => safeAnimate(el, { scale: 1.15 }, { duration: 0.15 })); el.addEventListener('mouseleave', () => safeAnimate(el, { scale: 1 }, { duration: 0.15 })); }); document.querySelectorAll('.repoCard').forEach(el => { el.addEventListener('mouseenter', () => safeAnimate(el, { scale: 1.08 }, { duration: 0.2 })); el.addEventListener('mouseleave', () => safeAnimate(el, { scale: 1 }, { duration: 0.2 })); }); } document.addEventListener('DOMContentLoaded', () => { loadAboutMe(); loadRepos().then(enableHoverAnimations); });
